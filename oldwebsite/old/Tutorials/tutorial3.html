function hist3d(data,binsizex,binsizey)
    xmax = maximum(data[:,1])
    xmin = minimum(data[:,1])
    ymax = maximum(data[:,2])
    ymin = minimum(data[:,2])
    # X grid point locations
    xpoints = linspace(xmin, xmax, binsizex+1)
    ypoints = linspace(ymin, ymax, binsizey+1)
    # Bins
    bins = zeros(binsizex,binsizey)
    # Bin allocation
    for i = 1:size(data)[1]
        tmp1 = data[i,1]*ones(binsizex,1) - xpoints[1:(end-1),1]
        [tmp1[i]=(tmp1[i] >= 0 ? 1 : 0) for i in 1:binsizex]
        xbinindx = Int(sum(tmp1))
        # print(tmp1)
        if xbinindx==0
            error("Zero in column 1 not possible")
        end
        tmp2 = data[i,2]*ones(binsizey,1) - ypoints[1:(end-1),1]
        [tmp2[i]=(tmp2[i] >= 0 ? 1 : 0) for i in 1:binsizey]
        ybinindx = Int(sum(tmp2))
        if ybinindx==0
            error("Zero in column 2 not possible")
        end
        bins[xbinindx,ybinindx] = bins[xbinindx,ybinindx] + 1
    end
    return bins , xpoints, ypoints
end
# target un-normalised density is a Gaussian
mu_t = [0.0;0.0]
sig_t = [3.0 0;0 3.0]
f1(x,y) = (1/sqrt((2*pi)^2*9))*exp(-0.5*([x;y]-mu_t)'*inv(sig_t)*([x;y]-mu_t))
# Proposal density is a Gaussian
# Generate samples from a Gaussian Distribution
f2(x,y) = rand(Distributions.MvNormal([x,y], [1.5 .0; .0 1.5]), 1)
# Initial state for Markov Chain
x_start = [3,3]
samples = Float64[]
N = 50000
for i = 1:N
    # First choose a jump state
    x_jump = map(f2,x_start[1],x_start[2])
    # Evaluate likelihood ratio
    lr = map(f1,x_jump[1],x_jump[2])/map(f1,x_start[1],x_start[2])
    #
    if lr >= 1
        x_start = x_jump
    else
        x_start = rand(Uniform(0,1), 1)[1] <= lr ? x_jump : x_start
    end
    if i > 3*N/4
        samples = [samples;x_start']
    end
end

mean_x1 = sum(samples[:,1])/size(samples)[1]
sigma_x1 = (1/(size(samples)[1]-1))*sum((samples[:,1]-mean_x1*ones(size(samples)[1])).^2)
mean_x2 = sum(samples[:,2])/size(samples)[1]
sigma_x2 = (1/(size(samples)[1]-1))*sum((samples[:,2]-mean_x2*ones(size(samples)[1])).^2)
sigma_x2 = (1/(size(samples)[1]-1))*sum((samples[:,1]-mean_x1*ones(size(samples)[1])).*(samples[:,2]-mean_x2*ones(size(samples)[1])))
